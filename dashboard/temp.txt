import { notFound } from "next/navigation";
import { Project } from "@/types";
import { getProject } from "@/lib/api";
import { getProjectEpisodes } from "@/lib/api";
import { Episode } from "@/types";
import SearchBar from "@/components/SearchBar";
import BranchFilter from "@/components/BranchFilter";
import EpisodeTimeline as ProjectEpisodeTimeline from "@/components/EpisodeTimeline";
import LoadingSkeleton from "@/components/LoadingSkeleton";

export default async function ProjectPage({
  params
}: {
  params: { projectId: string }
}) {
  const { projectId } = params;

  let project: Project;
  let episodes: Episode[] = [];

  try {
    project = await getProject(projectId);
    episodes = await getProjectEpisodes(projectId);
  } catch (error) {
    console.error(`Failed to load project ${projectId}:`, error);
    notFound();
  }

  return (
    <div className="space-y-8">
      <div className="flex items-center justify-between mb-6">
        <h1 className="text-2xl font-bold text-gray-900 dark:text-gray-100">
          {project.name}
        </h1>
        <div className="flex items-center space-x-3">
          <span className="px-3 py-1 bg-blue-100 dark:bg-blue-900/20 text-blue-800 dark:text-blue-200 rounded-full text-sm">
            {project.defaultBranch}
          </span>
          <a
            href="/settings"
            className="text-sm text-gray-600 dark:text-gray-400 hover:text-gray-900 dark:hover:text-gray-100"
          >
            Settings
          </a>
        </div>
      </div>

      <div className="space-y-4">
        <SearchBar projectId={projectId} />
        <BranchFilter projectId={projectId} />
      </div>

      <div className="space-y-6">
        <h2 className="text-xl font-semibold text-gray-900 dark:text-gray-100">
          Timeline
        </h2>
        {episodes.length === 0 ? (
          <div className="text-center py-8">
            <p className="text-gray-500 dark:text-gray-400">
              No episodes yet for this project
            </p>
          </div>
        ) : (
          <div className="space-y-4">
            {episodes.map((episode) => (
              <ProjectEpisodeTimeline
                key={episode.id}
                episodes={[episode]}
                showProjectName={false}
              />
            ))}
          </div>
        )}
      </div>
    </div>
  );
}

// Modified EpisodeTimeline component for project page that accepts episodes prop
function EpisodeTimeline({
  episodes,
  showProjectName = true
}: {
  episodes: Episode[];
  showProjectName?: boolean;
}) {
  if (episodes.length === 0) {
    return (
      <div className="text-center py-8">
        <p className="text-gray-500 dark:text-gray-400">
          No episodes to show
        </p>
      </div>
    );
  }

  return (
    <div className="space-y-3">
      {episodes.map((episode) => (
        <div key={episode.id} className="border border-gray-200 dark:border-gray-700 rounded-lg overflow-hidden">
          <div className="p-4">
            <div className="flex items-start space-x-3">
              {showProjectName && (
                <div className="flex-shrink-0">
                  <div className="w-8 h-8 flex items-center justify-center bg-blue-500 rounded-full">
                    <span className="text-white text-xs font-bold">
                      {episode.branchName.slice(0, 2).toUpperCase()}
                    </span>
                  </div>
                </div>
              )}
              <div>
                <h3 className="text-lg font-medium text-gray-900 dark:text-gray-100 mb-1">
                  {episode.label || `Session on ${episode.branchName}`}
                </h3>
                <p className="text-sm text-gray-500 dark:text-gray-400 mb-2">
                  {new Date(episode.startedAt).toLocaleString()}
                </p>
                {episode.changedFiles.length > 0 && (
                  <div className="flex flex-wrap gap-1 mb-2">
                    {episode.changedFiles.map((file) => (
                      <span key={file} className="px-2 py-0.5 bg-gray-100 dark:bg-gray-800/50 rounded text-xs">
                        {file.split('/').pop() || file}
                      </span>
                    ))}
                  </div>
                )}
                <div className="text-xs text-gray-500 dark:text-gray-400">
                  {episode.callCount} interactions • {
                    episode.endedAt
                      ? new Date(episode.endedAt).toLocaleString()
                      : 'Active'
                  }
                </div>
              </div>
            </div>
          </div>

          {/* Expandable content */}
          <div className="mt-4 pt-3 border-t border-gray-100 dark:border-gray-700">
            <button
              className={`w-full text-left flex items-center space-x-2 text-sm font-medium
                ${episode.episodeSummary || episode.explainDiffSummary ? 'text-gray-600 dark:text-gray-400 hover:text-gray-900 dark:hover:text-gray-100' : 'text-gray-400 dark:text-gray-500'}
              `}
            >
              <span>View details</span>
              <svg className="w-4 h-4 transition-transform" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2}
                  d="M19 9l-7 7-7-7"
                />
              </svg>
            </button>
            {/* In a real implementation, this would toggle visibility of detailed content */}
          </div>
        </div>
      ))
    }
    )
  );
}
